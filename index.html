// Initialize SecurePay SDK
import SecurePay from '@securepay/sdk';

const client = new SecurePay({
  apiKey: process.env.SP_SECRET_KEY,
  env: 'production'
});

// Create a payment order
const order = await client.orders.create({
  amount:   49900,   // in paise
  currency: 'INR',
  method:   'upi',
  capture:  true
});

// Handle settlement webhook
app.post('/webhook', (req, res) => {
  const { event, data } = req.body;
  if (event === 'payment.settled') {
    // ✅ T+0 — funds credited
    creditMerchant(data);
  }
  res.sendStatus(200);
});
FAQ

Common Questions

How fast is onboarding?
Typically 24–48 hours after KYC document verification — PAN, Aadhaar, and bank details. Submit once, go live immediately after approval.
What payment methods are supported?
UPI, credit & debit cards, wallets, net banking, QR payments, POS terminals, and Soundbox — all from one merchant dashboard.
Are there any hidden fees?
None. Transparent per-transaction pricing only. No setup fee, no monthly fee. Full pricing available on request during onboarding.
When are settlements credited?
Settlements processed daily, credited within 24 hours. T+0 same-day settlement available depending on your business configuration.
Is there a developer sandbox?
Yes — Dev Studio lets you simulate transactions, test edge cases, and validate your full integration before going live. No real money involved.
Is SecurePay RBI compliant?
Fully. SecurePay operates under all applicable RBI and NPCI guidelines, holds PCI-DSS certification with 256-bit encryption across all data flows.
